#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(SOURCES ${ROOT}/sources)
SET(LIBRARYLOADER ${ROOT}/../essentials/vilibraryloader)

#Include header directory
INCLUDE_DIRECTORIES(${LIBRARYLOADER})

#Initiate source files
FILE(GLOB VI_LIBRARYLOADER_HEADERS ${LIBRARYLOADER}/*.h)
FILE(GLOB VI_MAIN_SOURCES ${SOURCES}/*.cpp)

#Create executable
ADD_EXECUTABLE(visore ${VI_LIBRARYLOADER_HEADERS} ${VI_MAIN_SOURCES})

#Link Visore to Qt
TARGET_LINK_LIBRARIES(visore ${QT_LIBRARIES})

#Install file
INSTALL(TARGETS visore DESTINATION .)
