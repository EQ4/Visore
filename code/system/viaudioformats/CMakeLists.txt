#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_AUDIOFORMATS_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_AUDIOFORMATS_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(viaudioformats SHARED ${VI_AUDIOFORMATS_HEADERS} ${VI_AUDIOFORMATS_SOURCES})
TARGET_LINK_LIBRARIES(viaudioformats ${QT_LIBRARIES})

#Install file
INSTALL(TARGETS viaudioformats DESTINATION system)
