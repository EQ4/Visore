#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(CONNECTIONHEADERS ${ROOT}/../headers)
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS})

#Initiate source files
FILE(GLOB VI_BASSCONNECTION_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_BASSCONNECTION_HEADERS_ALL ${HEADERS}/*.h ${CONNECTIONHEADERS}/*.h)
SET(VI_BASSCONNECTION_HEADERS)
SET(VI_BASSCONNECTION_MOC_HEADERS)
FOREACH(HEADER ${VI_BASSCONNECTION_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_BASSCONNECTION_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_BASSCONNECTION_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT4_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_BASSCONNECTION_MOC_HEADERS})

#Create library
ADD_LIBRARY(vibassconnection SHARED  ${VI_BASSCONNECTION_HEADERS} ${VI_WRAPPER_HEADERS} ${VI_BASSCONNECTION_SOURCES})
ADD_LIBRARY(libbass SHARED IMPORTED)
SET_TARGET_PROPERTIES(libbass PROPERTIES IMPORTED_LOCATION ${VI_BASS_LIBRARY_INSTALL})
TARGET_LINK_LIBRARIES(vibassconnection ${QT_LIBRARIES} viessentials viaudioformats libbass)

#Install file
INSTALL(TARGETS vibassconnection DESTINATION system/audioengine/audioconnections)
