#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_WIDGET_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_WIDGET_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_WIDGET_HEADERS)
SET(VI_WIDGET_MOC_HEADERS)
FOREACH(HEADER ${VI_WIDGET_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_WIDGET_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_WIDGET_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT5_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_WIDGET_MOC_HEADERS})

#Wrap Qt files
QT5_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_WIDGET_MOC_HEADERS})

#Create library
ADD_LIBRARY(viwidget SHARED ${VI_WRAPPER_HEADERS} ${VI_WIDGET_SOURCES})
TARGET_LINK_LIBRARIES(viwidget ${QT_LIBRARIES} viaudioengine vitheme)

#Install file
INSTALL(TARGETS viwidget DESTINATION interface)
