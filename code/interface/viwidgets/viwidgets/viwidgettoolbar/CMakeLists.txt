#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS})

#Initiate source files
FILE(GLOB VI_WIDGETTOOLBAR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_WIDGETTOOLBAR_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_WIDGETTOOLBAR_HEADERS)
SET(VI_WIDGETTOOLBAR_MOC_HEADERS)
FOREACH(HEADER ${VI_WIDGETTOOLBAR_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_WIDGETTOOLBAR_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_WIDGETTOOLBAR_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
FILE(GLOB VI_ORANGETHEME2_RESOURCES ${INTERFACE}/vithemes/vithemes/viorangetheme/resources/*.qrc)
QT4_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_WIDGETTOOLBAR_MOC_HEADERS})
QT4_ADD_RESOURCES(VI_WRAPPER_RESOURCES ${VI_ORANGETHEME2_RESOURCES})

#Create library
ADD_LIBRARY(viwidgettoolbar SHARED ${VI_WRAPPER_HEADERS} ${VI_WIDGETTOOLBAR_SOURCES} ${VI_WRAPPER_RESOURCES})
TARGET_LINK_LIBRARIES(viwidgettoolbar ${QT_LIBRARIES} viwidget)

#Install file
INSTALL(TARGETS viwidgettoolbar DESTINATION interface)
