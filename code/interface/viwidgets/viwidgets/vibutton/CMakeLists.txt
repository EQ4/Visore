#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

#Initiate source files
FILE(GLOB VI_BUTTON_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_BUTTON_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_BUTTON_HEADERS)
SET(VI_BUTTON_MOC_HEADERS)
FOREACH(HEADER ${VI_BUTTON_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_BUTTON_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_BUTTON_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT4_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_BUTTON_MOC_HEADERS})

#Create library
ADD_LIBRARY(vibutton SHARED ${VI_WRAPPER_HEADERS} ${VI_BUTTON_SOURCES})
TARGET_LINK_LIBRARIES(vibutton ${QT_LIBRARIES} viwidget)

#Install file
INSTALL(TARGETS vibutton DESTINATION interface)
