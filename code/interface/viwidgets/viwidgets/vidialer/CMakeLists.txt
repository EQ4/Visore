#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

#Initiate source files
FILE(GLOB VI_DIALER_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_DIALER_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_DIALER_HEADERS)
SET(VI_DIALER_MOC_HEADERS)
FOREACH(HEADER ${VI_DIALER_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_DIALER_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_DIALER_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT5_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_DIALER_MOC_HEADERS})

#Create library
ADD_LIBRARY(vidialer SHARED ${VI_WRAPPER_HEADERS} ${VI_DIALER_SOURCES})
TARGET_LINK_LIBRARIES(vidialer ${QT_LIBRARIES} viwidget vigradientcreator)

#Install file
INSTALL(TARGETS vidialer DESTINATION ${VI_WIDGETSS_LOCATION})
