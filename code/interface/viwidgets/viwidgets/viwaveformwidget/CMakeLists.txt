#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS})

#Initiate source files
FILE(GLOB VI_WAVEFORMWIDGET_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_WAVEFORMWIDGET_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_WAVEFORMWIDGET_HEADERS)
SET(VI_WAVEFORMWIDGET_MOC_HEADERS)
FOREACH(HEADER ${VI_WAVEFORMWIDGET_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_WAVEFORMWIDGET_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_WAVEFORMWIDGET_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT5_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_WAVEFORMWIDGET_MOC_HEADERS})

#Create library
ADD_LIBRARY(viwaveformwidget SHARED ${VI_WRAPPER_HEADERS} ${VI_WAVEFORMWIDGET_SOURCES})
TARGET_LINK_LIBRARIES(viwaveformwidget ${QT_LIBRARIES} viwidget viwidgettoolbar)

#Install file
INSTALL(TARGETS viwaveformwidget DESTINATION ${VI_WIDGETSS_LOCATION})
