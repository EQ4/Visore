#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(FORMS ${ROOT}/forms)
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

#Initiate source files
FILE(GLOB VI_VOLUMEDIALWIDGET_FORMS ${FORMS}/*.ui)
FILE(GLOB VI_VOLUMEDIALWIDGET_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_VOLUMEDIALWIDGET_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_VOLUMEDIALWIDGET_HEADERS)
SET(VI_VOLUMEDIALWIDGET_MOC_HEADERS)
FOREACH(HEADER ${VI_VOLUMEDIALWIDGET_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_VOLUMEDIALWIDGET_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_VOLUMEDIALWIDGET_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT4_WRAP_UI(VI_WRAPPER_FORMS ${VI_VOLUMEDIALWIDGET_FORMS})
QT4_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_VOLUMEDIALWIDGET_MOC_HEADERS})

#Create library
ADD_LIBRARY(vivolumedialwidget SHARED ${VI_WRAPPER_HEADERS} ${VI_WRAPPER_FORMS} ${VI_VOLUMEDIALWIDGET_SOURCES})
TARGET_LINK_LIBRARIES(vivolumedialwidget ${QT_LIBRARIES} viwidget vibutton vidialer)

#Install file
INSTALL(TARGETS vivolumedialwidget DESTINATION ${VI_WIDGETSS_LOCATION})
