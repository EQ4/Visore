#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(FORMS ${ROOT}/forms)
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

#Initiate source files
FILE(GLOB VI_INPUTWIDGET_FORMS ${FORMS}/*.ui)
FILE(GLOB VI_INPUTWIDGET_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_INPUTWIDGET_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_INPUTWIDGET_HEADERS)
SET(VI_INPUTWIDGET_MOC_HEADERS)
FOREACH(HEADER ${VI_INPUTWIDGET_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_INPUTWIDGET_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_INPUTWIDGET_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT4_WRAP_UI(VI_WRAPPER_FORMS ${VI_INPUTWIDGET_FORMS})
QT4_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_INPUTWIDGET_MOC_HEADERS})

#Create library
ADD_LIBRARY(viinputwidget SHARED ${VI_WRAPPER_HEADERS} ${VI_INPUTWIDGET_SOURCES} ${VI_WRAPPER_FORMS})
TARGET_LINK_LIBRARIES(viinputwidget ${QT_LIBRARIES} vitheme viwidget vibutton)

#Install file
INSTALL(TARGETS viinputwidget DESTINATION ${VI_WIDGETSS_LOCATION})
