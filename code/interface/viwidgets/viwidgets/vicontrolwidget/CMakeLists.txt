#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(FORMS ${ROOT}/forms)
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

#Initiate source files
FILE(GLOB VI_CONTROLWIDGET_FORMS ${FORMS}/*.ui)
FILE(GLOB VI_CONTROLWIDGET_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_CONTROLWIDGET_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_CONTROLWIDGET_HEADERS)
SET(VI_CONTROLWIDGET_MOC_HEADERS)
FOREACH(HEADER ${VI_CONTROLWIDGET_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_CONTROLWIDGET_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_CONTROLWIDGET_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT4_WRAP_UI(VI_WRAPPER_FORMS ${VI_CONTROLWIDGET_FORMS})
QT4_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_CONTROLWIDGET_MOC_HEADERS})

#Create library
ADD_LIBRARY(vicontrolwidget SHARED ${VI_WRAPPER_HEADERS} ${VI_CONTROLWIDGET_SOURCES} ${VI_WRAPPER_FORMS})
TARGET_LINK_LIBRARIES(vicontrolwidget ${QT_LIBRARIES} vitheme vivolumebarwidget viplaybackwidget)

#Install file
INSTALL(TARGETS vicontrolwidget DESTINATION ${VI_WIDGETSS_LOCATION})
