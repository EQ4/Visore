#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)
SET(RESOURCES ${ROOT}/resources)

#Initiate source files
FILE(GLOB VI_VULCAN_THEME_RESOURCES ${RESOURCES}/*/*.qrc)
FILE(GLOB VI_VULCAN_THEME_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_VULCAN_THEME_HEADERS ${HEADERS}/*.h)

#Create library
QT5_ADD_RESOURCES(VI_RESOURCES ${VI_VULCAN_THEME_RESOURCES})
ADD_LIBRARY(vivulcanotheme SHARED ${VI_VULCAN_THEME_HEADERS} ${VI_VULCAN_THEME_SOURCES} ${VI_RESOURCES})
TARGET_LINK_LIBRARIES(vivulcanotheme vitheme)
QT5_USE_MODULES(vivulcanotheme Gui)

#Install file
INSTALL(TARGETS vivulcanotheme DESTINATION interface/themes)
