#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)
SET(RESOURCES ${ROOT}/resources)

#Initiate source files
FILE(GLOB VI_ORANGETHEME_RESOURCES ${RESOURCES}/*.qrc)
FILE(GLOB VI_THEME_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_THEME_HEADERS ${HEADERS}/*.h)

#Create library
QT5_ADD_RESOURCES(VI_RESOURCES ${VI_ORANGETHEME_RESOURCES})
ADD_LIBRARY(vitheme SHARED ${VI_THEME_HEADERS} ${VI_THEME_SOURCES} ${VI_RESOURCES})
TARGET_LINK_LIBRARIES(vitheme viessentials)

#Install file
INSTALL(TARGETS vitheme DESTINATION interface)
