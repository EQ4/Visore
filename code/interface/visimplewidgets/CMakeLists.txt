#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(FORMS ${ROOT}/forms)
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_SIMPLEWIDGETS_FORMS ${FORMS}/*.ui ${FORMS}/*/*.ui)
FILE(GLOB VI_SIMPLEWIDGETS_SOURCES ${SOURCES}/*.cpp ${SOURCES}/*/*.cpp)
FILE(GLOB VI_SIMPLEWIDGETS_HEADERS ${HEADERS}/*.h ${HEADERS}/*/*.h)

#Create library
QT5_WRAP_UI(VI_WRAPPER_FORMS ${VI_SIMPLEWIDGETS_FORMS})
ADD_LIBRARY(visimplewidgets SHARED ${VI_SIMPLEWIDGETS_HEADERS} ${VI_SIMPLEWIDGETS_SOURCES} ${VI_WRAPPER_FORMS})
TARGET_LINK_LIBRARIES(visimplewidgets vitheme)
QT5_USE_MODULES(visimplewidgets Widgets Multimedia Xml)

#Install file
INSTALL(TARGETS visimplewidgets DESTINATION interface)
