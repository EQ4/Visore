#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(FORMS ${ROOT}/forms)
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

#Initiate source files
FILE(GLOB VI_SIMPLEWIDGETS_FORMS ${FORMS}/*.ui)
FILE(GLOB VI_SIMPLEWIDGETS_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_SIMPLEWIDGETS_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_SIMPLEWIDGETS_HEADERS)
SET(VI_SIMPLEWIDGETS_MOC_HEADERS)
FOREACH(HEADER ${VI_SIMPLEWIDGETS_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_SIMPLEWIDGETS_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_SIMPLEWIDGETS_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT5_WRAP_UI(VI_WRAPPER_FORMS ${VI_SIMPLEWIDGETS_FORMS})
QT5_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_SIMPLEWIDGETS_MOC_HEADERS})

#Create library
ADD_LIBRARY(visimplewidgets SHARED ${VI_WRAPPER_HEADERS} ${VI_SIMPLEWIDGETS_SOURCES} ${VI_WRAPPER_FORMS})
TARGET_LINK_LIBRARIES(visimplewidgets ${QT_LIBRARIES} vitheme)

#Install file
INSTALL(TARGETS visimplewidgets DESTINATION interface)
