#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(FORMS ${ROOT}/forms)
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers. Required for the auto-created ui headers
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_BINARY_DIR})

#Initiate source files
FILE(GLOB VI_MAINWINDOW_FORMS ${FORMS}/*.ui)
FILE(GLOB VI_MAINWINDOW_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_MAINWINDOW_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_MAINWINDOW_HEADERS)
SET(VI_MAINWINDOW_MOC_HEADERS)
FOREACH(HEADER ${VI_MAINWINDOW_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_MAINWINDOW_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_MAINWINDOW_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT4_WRAP_UI(VI_WRAPPER_FORMS ${VI_MAINWINDOW_FORMS})
QT4_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_MAINWINDOW_MOC_HEADERS})

#Create library
ADD_LIBRARY(vimainwindow SHARED ${VI_WRAPPER_FORMS} ${VI_WRAPPER_HEADERS} ${VI_MAINWINDOW_SOURCES})
TARGET_LINK_LIBRARIES(vimainwindow ${QT_LIBRARIES} vitheme viwaveformwidget)

#Install file
INSTALL(TARGETS vimainwindow DESTINATION interface)
