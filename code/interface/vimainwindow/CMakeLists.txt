#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(FORMS ${ROOT}/forms)
SET(RESOURCES ${ROOT}/resources)
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_MAINWINDOW_HEADERS ${HEADERS}/*.h)
FILE(GLOB VI_MAINWINDOW_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_MAINWINDOW_FORMS ${FORMS}/*.ui)
SET(VI_MAINWINDOW_RESOURCES ${RESOURCES}/viresources.qrc)
FILE(GLOB VI_MAINWINDOW_MOC_HEADERS

)

#Wrap Qt files
QT4_WRAP_UI(VI_WRAPPER_FORMS ${VI_MAINWINDOW_FORMS})
QT4_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_MAINWINDOW_MOC_HEADERS})
QT4_ADD_RESOURCES(VI_WRAPPER_RESOURCES ${VI_MAINWINDOW_RESOURCES})

#Create library
ADD_LIBRARY(vimainwindow SHARED ${VI_WRAPPER_FORMS} ${VI_WRAPPER_HEADERS} ${VI_WRAPPER_RESOURCES} ${VI_MAINWINDOW_SOURCES})
TARGET_LINK_LIBRARIES(vimainwindow ${QT_LIBRARIES})

#Install file
INSTALL(TARGETS vimainwindow DESTINATION interface)
