#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_CROSS_CORRELATOR_HEADERS ${HEADERS}/*.h)
FILE(GLOB VI_CROSS_CORRELATOR_SOURCES ${SOURCES}/*.cpp)

#Create library
ADD_LIBRARY(vicrosscorrelator SHARED ${VI_CROSS_CORRELATOR_HEADERS} ${VI_CROSS_CORRELATOR_SOURCES})
TARGET_LINK_LIBRARIES(vicrosscorrelator vicorrelator)
QT5_USE_MODULES(vicrosscorrelator Multimedia Xml)

#Install file
INSTALL(TARGETS vicrosscorrelator DESTINATION ${VI_CORRELATOR_INSTALL_PATH})
