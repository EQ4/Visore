#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_AVERAGE_ACTIVATION_FUNCTION_HEADERS ${HEADERS}/*.h)
FILE(GLOB VI_AVERAGE_ACTIVATION_FUNCTION_SOURCES ${SOURCES}/*.cpp)

#Create library
ADD_LIBRARY(viaverageactivationfunction SHARED ${VI_AVERAGE_ACTIVATION_FUNCTION_HEADERS} ${VI_AVERAGE_ACTIVATION_FUNCTION_SOURCES})

#Install file
INSTALL(TARGETS viaverageactivationfunction DESTINATION ${VI_ACTIVATION_FUNCTION_INSTALL_PATH})
