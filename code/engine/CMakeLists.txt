#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(VI_AUDIOFORMATS ${ROOT}/viaudioformats)
SET(VI_AUDIOCONNECTIONS ${ROOT}/viaudioconnections)
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate sub-directories
ADD_SUBDIRECTORY(${VI_AUDIOFORMATS})
ADD_SUBDIRECTORY(${VI_AUDIOCONNECTIONS})

#Initiate source files
FILE(GLOB VI_AUDIOENGINE_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_AUDIOENGINE_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_AUDIOENGINE_HEADERS)
SET(VI_AUDIOENGINE_MOC_HEADERS)
FOREACH(HEADER ${VI_AUDIOENGINE_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_AUDIOENGINE_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_AUDIOENGINE_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT4_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_AUDIOENGINE_MOC_HEADERS})

#Create library
ADD_LIBRARY(viaudioengine SHARED ${VI_AUDIOENGINE_HEADERS} ${VI_AUDIOENGINE_SOURCES} ${VI_WRAPPER_HEADERS})
TARGET_LINK_LIBRARIES(viaudioengine ${QT_LIBRARIES} viessentials)

#Install file
INSTALL(TARGETS viaudioengine DESTINATION engine)
