#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_PROCESSINGCHAIN_SOURCES ${SOURCES}/*.cpp ${SOURCES}/*/*.cpp ${SOURCES}/*/*/*.cpp)
FILE(GLOB VI_PROCESSINGCHAIN_HEADERS_ALL ${HEADERS}/*.h ${HEADERS}/*/*.h ${HEADERS}/*/*/*.h)
SET(VI_PROCESSINGCHAIN_HEADERS)
SET(VI_PROCESSINGCHAIN_MOC_HEADERS)
FOREACH(HEADER ${VI_PROCESSINGCHAIN_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_PROCESSINGCHAIN_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_PROCESSINGCHAIN_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT5_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_PROCESSINGCHAIN_MOC_HEADERS})

#Create library
ADD_LIBRARY(viprocessingchain SHARED ${VI_PROCESSINGCHAIN_HEADERS} ${VI_PROCESSINGCHAIN_SOURCES} ${VI_WRAPPER_HEADERS})
TARGET_LINK_LIBRARIES(viprocessingchain ${QT_QTCORE_LIBRARIES} viessentials viaudioconnection)

#Install file
INSTALL(TARGETS viprocessingchain DESTINATION engine)
