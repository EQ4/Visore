#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_WINDOWER_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_WINDOWER_HEADERS ${HEADERS}/*.h)

#Add C++ macro
ADD_DEFINITIONS(-DVIWINDOWERSLOCATION="${VI_WINDOWERS_LOCATION}")

#Create library
ADD_LIBRARY(viwindower SHARED ${VI_WINDOWER_HEADERS} ${VI_WINDOWER_SOURCES})
TARGET_LINK_LIBRARIES(viwindower ${QT_LIBRARIES} viessentials)

#Install file
INSTALL(TARGETS viwindower DESTINATION engine)
