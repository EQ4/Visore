#Setup directory paths

SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers

#Initiate source files
FILE(GLOB VI_WAVEFORMAT_SOURCES ${SOURCES}/*.cpp)

#Create library
ADD_LIBRARY(viwaveformat SHARED ${VI_WAVEFORMAT_HEADERS} ${VI_WAVEFORMAT_SOURCES})
TARGET_LINK_LIBRARIES(viwaveformat ${QT_LIBRARIES})

#Install file
INSTALL(TARGETS viwaveformat DESTINATION ${VI_AUDIOFORMATS_LOCATION})
