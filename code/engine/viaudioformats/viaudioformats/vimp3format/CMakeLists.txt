#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS})

#Initiate source files
FILE(GLOB VI_MP3FORMAT_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_MP3FORMAT_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vimp3format SHARED ${VI_MP3FORMAT_HEADERS} ${VI_MP3FORMAT_SOURCES})
TARGET_LINK_LIBRARIES(vimp3format ${QT_LIBRARIES})

#Install file
INSTALL(TARGETS vimp3format DESTINATION ${VI_AUDIOFORMATS_LOCATION})
