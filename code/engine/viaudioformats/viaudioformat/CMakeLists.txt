#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_AUDIOFORMAT_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_AUDIOFORMAT_HEADERS ${HEADERS}/*.h)

#Add C++ macro
ADD_DEFINITIONS(-DVIAUDIOFORMATSLOCATION="${VI_AUDIOFORMATS_LOCATION}")

#Create library
ADD_LIBRARY(viaudioformat SHARED ${VI_AUDIOFORMAT_HEADERS} ${VI_AUDIOFORMAT_SOURCES})
TARGET_LINK_LIBRARIES(viaudioformat ${QT_LIBRARIES} viessentials)

#Install file
INSTALL(TARGETS viaudioformat DESTINATION engine)
