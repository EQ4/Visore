#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)
SET(FFTREAL_HEADERS ${EXTERNAL}/fftreal/headers)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS} ${FFTREAL_HEADERS})

#Initiate source files
FILE(GLOB VI_SPECTRUMANALYZER_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_SPECTRUMANALYZER_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_SPECTRUMANALYZER_HEADERS)
SET(VI_SPECTRUMANALYZER_MOC_HEADERS)
FOREACH(HEADER ${VI_SPECTRUMANALYZER_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_SPECTRUMANALYZER_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_SPECTRUMANALYZER_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT5_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_SPECTRUMANALYZER_MOC_HEADERS})

#Create library
ADD_LIBRARY(vispectrumanalyzer SHARED ${VI_SPECTRUMANALYZER_SOURCES} ${VI_WRAPPER_HEADERS})
TARGET_LINK_LIBRARIES(vispectrumanalyzer ${QT_LIBRARIES} viwindowfunction viwaveformer fftreal)

#Install file
INSTALL(TARGETS vispectrumanalyzer DESTINATION engine)
