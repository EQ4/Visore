#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_WINDOWFUNCTION_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_WINDOWFUNCTION_HEADERS ${HEADERS}/*.h)

#Add C++ macro
ADD_DEFINITIONS(-DVIWINDOWFUNCTIONSLOCATION="${VI_WINDOWFUNCTIONS_LOCATION}")

#Create library
ADD_LIBRARY(viwindowfunction SHARED ${VI_WINDOWFUNCTION_HEADERS} ${VI_WINDOWFUNCTION_SOURCES})
TARGET_LINK_LIBRARIES(viwindowfunction ${QT_LIBRARIES} viessentials)

#Install file
INSTALL(TARGETS viwindowfunction DESTINATION engine)
