#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS})

#Initiate source files
FILE(GLOB VI_HANNWINDOWFUNCTION_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_HANNWINDOWFUNCTION_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vihannwindowfunction SHARED ${VI_HANNWINDOWFUNCTION_HEADERS} ${VI_HANNWINDOWFUNCTION_SOURCES})
TARGET_LINK_LIBRARIES(vihannwindowfunction ${QT_LIBRARIES} viwindowfunction)

#Install file
INSTALL(TARGETS vihannwindowfunction DESTINATION ${VI_WINDOWFUNCTIONS_LOCATION})
