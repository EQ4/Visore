#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS})

#Initiate source files
FILE(GLOB VI_NOWINDOWFUNCTION_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_NOWINDOWFUNCTION_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vinowindowfunction SHARED ${VI_NOWINDOWFUNCTION_HEADERS} ${VI_NOWINDOWFUNCTION_SOURCES})
TARGET_LINK_LIBRARIES(vinowindowfunction ${QT_LIBRARIES})

#Install file
INSTALL(TARGETS vinowindowfunction DESTINATION ${VI_WINDOWFUNCTIONS_LOCATION})
