#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_LAGRANGE_PREDICTOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_LAGRANGE_PREDICTOR_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vilagrangepredictor SHARED ${VI_LAGRANGE_PREDICTOR_HEADERS} ${VI_LAGRANGE_PREDICTOR_SOURCES})
TARGET_LINK_LIBRARIES(vilagrangepredictor vipredictor)
QT5_USE_MODULES(vilagrangepredictor Xml Multimedia)

#Install file
INSTALL(TARGETS vilagrangepredictor DESTINATION ${VI_PREDICTORS_INSTALL_PATH})
