#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_LINEAR_PREDICTOR_HEADERS ${HEADERS}/*.h)
FILE(GLOB VI_LINEAR_PREDICTOR_SOURCES ${SOURCES}/*.cpp)

#Create library
ADD_LIBRARY(vilinearpredictor SHARED ${VI_LINEAR_PREDICTOR_HEADERS} ${VI_LINEAR_PREDICTOR_SOURCES})
TARGET_LINK_LIBRARIES(vilinearpredictor vipredictor)
QT5_USE_MODULES(vilinearpredictor Multimedia Xml)

#Install file
INSTALL(TARGETS vilinearpredictor DESTINATION ${VI_PREDICTOR_INSTALL_PATH})
