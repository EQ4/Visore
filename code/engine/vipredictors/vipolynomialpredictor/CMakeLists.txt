#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_POLYNOMIAL_PREDICTOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_POLYNOMIAL_PREDICTOR_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vipolynomialpredictor SHARED ${VI_POLYNOMIAL_PREDICTOR_HEADERS} ${VI_POLYNOMIAL_PREDICTOR_SOURCES})
TARGET_LINK_LIBRARIES(vipolynomialpredictor vipredictor)
QT5_USE_MODULES(vipolynomialpredictor Xml Multimedia)

#Install file
INSTALL(TARGETS vipolynomialpredictor DESTINATION ${VI_PREDICTORS_INSTALL_PATH})
