#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_FOURIER_PREDICTOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_FOURIER_PREDICTOR_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vifourierpredictor SHARED ${VI_FOURIER_PREDICTOR_HEADERS} ${VI_FOURIER_PREDICTOR_SOURCES})
TARGET_LINK_LIBRARIES(vifourierpredictor vipredictor)
QT5_USE_MODULES(vifourierpredictor Xml Multimedia)

#Install file
INSTALL(TARGETS vifourierpredictor DESTINATION ${VI_PREDICTORS_INSTALL_PATH})
