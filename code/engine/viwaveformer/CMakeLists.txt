#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS})

#Initiate source files
FILE(GLOB VI_WAVEFORMER_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_WAVEFORMER_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_WAVEFORMER_HEADERS)
SET(VI_WAVEFORMER_MOC_HEADERS)
FOREACH(HEADER ${VI_WAVEFORMER_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_WAVEFORMER_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_WAVEFORMER_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT5_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_WAVEFORMER_MOC_HEADERS})

#Create library
ADD_LIBRARY(viwaveformer SHARED ${VI_WAVEFORMER_SOURCES} ${VI_WRAPPER_HEADERS})
TARGET_LINK_LIBRARIES(viwaveformer ${QT_LIBRARIES} viessentials viaudioformat viaudiobuffer)

#Install file
INSTALL(TARGETS viwaveformer DESTINATION engine)
