#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_NEWTON_POLYNOMIAL_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_NEWTON_POLYNOMIAL_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vinewtonpolynomial SHARED ${VI_NEWTON_POLYNOMIAL_HEADERS} ${VI_NEWTON_POLYNOMIAL_SOURCES})
TARGET_LINK_LIBRARIES(vinewtonpolynomial vipolynomial)

#Install file
INSTALL(TARGETS vinewtonpolynomial DESTINATION ${VI_POLYNOMIALS_INSTALL_PATH})
