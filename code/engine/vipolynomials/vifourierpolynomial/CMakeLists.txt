#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_FOURIER_POLYNOMIAL_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_FOURIER_POLYNOMIAL_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vifourierpolynomial SHARED ${VI_FOURIER_POLYNOMIAL_HEADERS} ${VI_FOURIER_POLYNOMIAL_SOURCES})
TARGET_LINK_LIBRARIES(vifourierpolynomial vipolynomial)

#Install file
INSTALL(TARGETS vifourierpolynomial DESTINATION ${VI_POLYNOMIALS_INSTALL_PATH})
