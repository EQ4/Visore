#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_STANDARD_POLYNOMIAL_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_STANDARD_POLYNOMIAL_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vistandardpolynomial SHARED ${VI_STANDARD_POLYNOMIAL_HEADERS} ${VI_STANDARD_POLYNOMIAL_SOURCES})
TARGET_LINK_LIBRARIES(vistandardpolynomial vipolynomial)

#Install file
INSTALL(TARGETS vistandardpolynomial DESTINATION ${VI_POLYNOMIALS_INSTALL_PATH})
