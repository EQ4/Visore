#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS})

#Initiate source files
FILE(GLOB VI_AUDIOCONNECTION_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_AUDIOCONNECTION_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_AUDIOCONNECTION_HEADERS)
SET(VI_AUDIOCONNECTION_MOC_HEADERS)
FOREACH(HEADER ${VI_AUDIOCONNECTION_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_AUDIOCONNECTION_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_AUDIOCONNECTION_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT5_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_AUDIOCONNECTION_MOC_HEADERS})

#Create library
ADD_LIBRARY(viaudioconnection SHARED ${VI_WRAPPER_HEADERS} ${VI_AUDIOCONNECTION_SOURCES})
TARGET_LINK_LIBRARIES(viaudioconnection ${QT_LIBRARIES} viaudioformat viaudiobuffer)

#Install file
INSTALL(TARGETS viaudioconnection DESTINATION engine)
