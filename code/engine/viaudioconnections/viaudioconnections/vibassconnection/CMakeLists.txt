#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(BASS_HEADERS ${EXTERNAL}/bass/headers)
SET(BASS_PLUGIN_HEADERS ${EXTERNAL}/bass/headers/plugins)
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS} ${BASS_HEADERS} ${BASS_PLUGIN_HEADERS})

#Initiate source files
FILE(GLOB VI_BASSCONNECTION_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_BASSCONNECTION_HEADERS_ALL ${HEADERS}/*.h ${ENGINE}/viaudioconnections/viaudioconnection/headers/*.h)
SET(VI_BASSCONNECTION_HEADERS)
SET(VI_BASSCONNECTION_MOC_HEADERS)
FOREACH(HEADER ${VI_BASSCONNECTION_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_BASSCONNECTION_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_BASSCONNECTION_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT5_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_BASSCONNECTION_MOC_HEADERS})

#Add C++ macro
ADD_DEFINITIONS(-DBASSPLUGINSLOCATION="${BASS_PLUGINS_LOCATION}" -DBASSLIBRARYLOCATION="${BASS_LIBRARY_LOCATION}")

#Create library
ADD_LIBRARY(vibassconnection SHARED  ${VI_BASSCONNECTION_HEADERS} ${VI_WRAPPER_HEADERS} ${VI_BASSCONNECTION_SOURCES})
TARGET_LINK_LIBRARIES(vibassconnection ${QT_LIBRARIES} ${BASS_LIBRARY} viaudioconnection)
