#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS} ${QT_HEADERS} ${QT_PLUGIN_HEADERS} /home/visore/Qt5/qt5/qtbase/include/QtMultimedia)

#Initiate source files
FILE(GLOB VI_QTCONNECTION_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_QTCONNECTION_HEADERS_ALL ${HEADERS}/*.h ${ENGINE}/viaudioconnections/viaudioconnection/headers/*.h)
SET(VI_QTCONNECTION_HEADERS)
SET(VI_QTCONNECTION_MOC_HEADERS)
FOREACH(HEADER ${VI_QTCONNECTION_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_QTCONNECTION_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_QTCONNECTION_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT5_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_QTCONNECTION_MOC_HEADERS})

#Create library
ADD_LIBRARY(viqtconnection SHARED  ${VI_QTCONNECTION_HEADERS} ${VI_WRAPPER_HEADERS} ${VI_QTCONNECTION_SOURCES})
TARGET_LINK_LIBRARIES(viqtconnection ${Qt5Multimedia_LIBRARIES} viaudioconnection vicoder)

#Install file
INSTALL(TARGETS viqtconnection DESTINATION ${VI_AUDIOCONNECTIONS_LOCATION})
