#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_TRIG_INTERPOLATOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_TRIG_INTERPOLATOR_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vitriginterpolator SHARED ${VI_TRIG_INTERPOLATOR_HEADERS} ${VI_TRIG_INTERPOLATOR_SOURCES})
TARGET_LINK_LIBRARIES(vitriginterpolator viinterpolator)
QT5_USE_MODULES(vitriginterpolator Xml Multimedia)

#Install file
INSTALL(TARGETS vitriginterpolator DESTINATION ${VI_INTERPOLATORS_INSTALL_PATH})
