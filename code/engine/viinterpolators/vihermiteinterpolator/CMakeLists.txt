#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_HERMITE_INTERPOLATOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_HERMITE_INTERPOLATOR_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vihermiteinterpolator SHARED ${VI_HERMITE_INTERPOLATOR_HEADERS} ${VI_HERMITE_INTERPOLATOR_SOURCES})
TARGET_LINK_LIBRARIES(vihermiteinterpolator viinterpolator)
QT5_USE_MODULES(vihermiteinterpolator Xml Multimedia)

#Install file
INSTALL(TARGETS vihermiteinterpolator DESTINATION ${VI_INTERPOLATORS_INSTALL_PATH})
