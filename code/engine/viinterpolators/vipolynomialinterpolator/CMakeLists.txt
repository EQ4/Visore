#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_POLYNOMIAL_INTERPOLATOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_POLYNOMIAL_INTERPOLATOR_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vipolynomialinterpolator SHARED ${VI_POLYNOMIAL_INTERPOLATOR_HEADERS} ${VI_POLYNOMIAL_INTERPOLATOR_SOURCES})
TARGET_LINK_LIBRARIES(vipolynomialinterpolator viinterpolator)
QT5_USE_MODULES(vipolynomialinterpolator Xml Multimedia)

#Install file
INSTALL(TARGETS vipolynomialinterpolator DESTINATION ${VI_INTERPOLATORS_INSTALL_PATH})
