#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_CUBIC_INTERPOLATOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_CUBIC_INTERPOLATOR_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vicubicinterpolator SHARED ${VI_CUBIC_INTERPOLATOR_HEADERS} ${VI_CUBIC_INTERPOLATOR_SOURCES})
TARGET_LINK_LIBRARIES(vicubicinterpolator viinterpolator)
QT5_USE_MODULES(vicubicinterpolator Core Multimedia)

#Install file
INSTALL(TARGETS vicubicinterpolator DESTINATION ${VI_INTERPOLATORS_INSTALL_PATH})
