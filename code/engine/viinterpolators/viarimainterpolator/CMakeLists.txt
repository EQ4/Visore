#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

ADD_DEFINITIONS(-DX12ABUILD="${X12A_EXECUTABLE_BUILD}")
ADD_DEFINITIONS(-DX12AINSTALL="${X12A_EXECUTABLE_INSTALL}")

#Initiate source files
FILE(GLOB VI_ARIMA_INTERPOLATOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_ARIMA_INTERPOLATOR_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(viarimainterpolator SHARED ${VI_ARIMA_INTERPOLATOR_HEADERS} ${VI_ARIMA_INTERPOLATOR_SOURCES})
TARGET_LINK_LIBRARIES(viarimainterpolator viinterpolator vigretl)
QT5_USE_MODULES(viarimainterpolator Xml Multimedia)

#Install file
INSTALL(TARGETS viarimainterpolator DESTINATION ${VI_INTERPOLATORS_INSTALL_PATH})
