#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_NEWTON_INTERPOLATOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_NEWTON_INTERPOLATOR_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vinewtoninterpolator SHARED ${VI_NEWTON_INTERPOLATOR_HEADERS} ${VI_NEWTON_INTERPOLATOR_SOURCES})
TARGET_LINK_LIBRARIES(vinewtoninterpolator viinterpolator)
QT5_USE_MODULES(vinewtoninterpolator Xml Multimedia)

#Install file
INSTALL(TARGETS vinewtoninterpolator DESTINATION ${VI_INTERPOLATORS_INSTALL_PATH})
