#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_PRIMITIVE_INTERPOLATOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_PRIMITIVE_INTERPOLATOR_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(viprimitiveinterpolator SHARED ${VI_PRIMITIVE_INTERPOLATOR_HEADERS} ${VI_PRIMITIVE_INTERPOLATOR_SOURCES})
TARGET_LINK_LIBRARIES(viprimitiveinterpolator viinterpolator)
QT5_USE_MODULES(viprimitiveinterpolator Xml Multimedia)

#Install file
INSTALL(TARGETS viprimitiveinterpolator DESTINATION ${VI_INTERPOLATORS_INSTALL_PATH})
