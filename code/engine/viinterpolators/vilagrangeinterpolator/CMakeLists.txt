#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Initiate source files
FILE(GLOB VI_LAGRANGE_INTERPOLATOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_LAGRANGE_INTERPOLATOR_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vilagrangeinterpolator SHARED ${VI_LAGRANGE_INTERPOLATOR_HEADERS} ${VI_LAGRANGE_INTERPOLATOR_SOURCES})
TARGET_LINK_LIBRARIES(vilagrangeinterpolator viinterpolator)
QT5_USE_MODULES(vilagrangeinterpolator Xml Multimedia)

#Install file
INSTALL(TARGETS vilagrangeinterpolator DESTINATION ${VI_INTERPOLATORS_INSTALL_PATH})
