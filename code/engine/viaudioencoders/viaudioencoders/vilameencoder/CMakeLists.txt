#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS})

#Initiate source files
FILE(GLOB VI_LAMEENCODER_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_LAMEENCODER_HEADERS ${HEADERS}/*.h)

#Create library
ADD_LIBRARY(vilameencoder SHARED ${VI_LAMEENCODER_HEADERS} ${VI_LAMEENCODER_SOURCES})
TARGET_LINK_LIBRARIES(vilameencoder ${QT_LIBRARIES} viaudioencoder)

#Install file
INSTALL(TARGETS vilameencoder DESTINATION ${VI_AUDIOENCODERS_LOCATION})
