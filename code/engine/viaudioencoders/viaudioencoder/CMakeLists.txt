#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${HEADERS})

#Initiate source files
FILE(GLOB VI_AUDIOENCODER_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_AUDIOENCODER_HEADERS ${HEADERS}/*.h)

#Add C++ macro
ADD_DEFINITIONS(-DVIAUDIOENCODERSLOCATION="${VI_AUDIOENCODERS_LOCATION}")

#Create library
ADD_LIBRARY(viaudioencoder SHARED ${VI_AUDIOENCODER_HEADERS} ${VI_AUDIOENCODER_SOURCES})
TARGET_LINK_LIBRARIES(viaudioencoder ${QT_LIBRARIES} viessentials viaudioformat)

#Install file
INSTALL(TARGETS viaudioencoder DESTINATION engine)
