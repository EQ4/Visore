#Setup directory paths
SET(ROOT ${CMAKE_CURRENT_SOURCE_DIR})
SET(HEADERS ${ROOT}/headers)
SET(SOURCES ${ROOT}/sources)

#Include common headers
INCLUDE_DIRECTORIES(${EXTERNAL}/echonest/headers ${EXTERNAL}/qjson/headers)

#Initiate source files
FILE(GLOB VI_SONGDETECTOR_SOURCES ${SOURCES}/*.cpp)
FILE(GLOB VI_SONGDETECTOR_HEADERS_ALL ${HEADERS}/*.h)
SET(VI_SONGDETECTOR_HEADERS)
SET(VI_SONGDETECTOR_MOC_HEADERS)
FOREACH(HEADER ${VI_SONGDETECTOR_HEADERS_ALL})
    FILE(STRINGS "${HEADER}" lines REGEX "Q_OBJECT")
    IF(lines)
        LIST(APPEND VI_SONGDETECTOR_MOC_HEADERS "${HEADER}")
	ELSE()
		LIST(APPEND VI_SONGDETECTOR_HEADERS "${HEADER}")
    ENDIF()
ENDFOREACH()

#Wrap Qt files
QT4_WRAP_CPP(VI_WRAPPER_HEADERS ${VI_SONGDETECTOR_MOC_HEADERS})

#Create library
ADD_LIBRARY(visongdetector SHARED ${VI_SONGDETECTOR_HEADERS} ${VI_SONGDETECTOR_SOURCES} ${VI_WRAPPER_HEADERS})
TARGET_LINK_LIBRARIES(visongdetector ${QT_QTCORE_LIBRARIES} ${QT_QTNETWORK_LIBRARIES} ${ECHONEST_LIBRARY} qjson viwaveformer)

#Install file
INSTALL(TARGETS visongdetector DESTINATION engine)
